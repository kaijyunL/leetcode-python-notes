def setZeroes(matrix):
m, n = len(matrix), len(matrix[0])
flag_col0 = False
# 阶段一：用第一行/第一列作为标记，遇0则投影打标记
for i in range(m):
# 第一列由专属变量掌管，避免与第一行(matrix[0][0])冲突
if matrix[i][0] == 0:
flag_col0 = True
# 扫描剩余元素（第一列以外的）
for j in range(1, n):
if matrix[i][j] == 0:
matrix[i][0] = 0 # 第一列对应的行头打标记
matrix[0][j] = 0 # 第一行对应的列头打标记
# 阶段二：必须从下往上、从右往左反向修改，防止标记被自己提前覆盖清除
for i in range(m - 1, -1, -1):
for j in range(n - 1, 0, -1):
if matrix[i][0] == 0 or matrix[0][j] == 0:
matrix[i][j] = 0
# 最后根据 flag_col0 处理特殊的第一列
if flag_col0:
matrix[i][0] = 0
