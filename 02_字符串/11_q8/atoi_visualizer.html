LeetCode 8: 字符串转换整数 (atoi)

solution.py
class Solution:
def myAtoi(self, s: str) -> int:
# 1. 处理前导空格
s = s.lstrip()
if not s:
return 0
# 2. 初始化变量
res, i, sign = 0, 0, 1
int_max, int_min = 2147483647, -2147483648
# 3. 处理符号位
if s[0] == '-':
sign = -1
i = 1
elif s[0] == '+':
i = 1
# 4. 遍历数字并转换
while i < len(s) and s[i].isdigit():
res = res * 10 + int(s[i])
i += 1
# 实时越界检查
if sign * res > int_max:
return int_max
if sign * res < int_min:
return int_min
# 5. 返回结果
return sign * res
