Z 字形变换 - 核心逻辑动画执行仪 (解法二)

按行排序：通过一个游标 i 和方向 flag 决定当前字符落入哪一行

当前字符 ch ''
行指针 i 0
方向 flag -1 
class Solution:


    def convert(self, s: str, numRows: int) -> str:


        if numRows == 1 or numRows >= len(s):


            return s


            


        rows = [''] * numRows


        i, flag = 0, -1


        


        for ch in s:


            rows[i] += ch


            if i == 0 or i == numRows - 1:


                flag = -flag


            i += flag


            


        return "".join(rows)
